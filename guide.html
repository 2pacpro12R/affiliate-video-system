ÉTAPE 02 Installer FFmpeg ~5 min
🎞️
C'est quoi FFmpeg ?
💡 FFmpeg = le moteur invisible qui fusionne ta vidéo Remotion + ta voix off ElevenLabs en un seul MP4 final. Tu ne l'ouvres jamais, il tourne en coulisses.
⚡
Installation via winget (le plus simple)
Dans PowerShell, tape cette commande — winget est intégré dans Windows 10/11 :
winget install ffmpegCOPIER
✅ Accepte les conditions si demandé. L'installation prend 1-2 minutes.
🔄
Si winget ne fonctionne pas
Télécharge manuellement sur ffmpeg.org → Builds → Windows → essentials. Extrait le ZIP dans C:\ffmpeg\ et ajoute C:\ffmpeg\bin au PATH Windows.
💬 Dit-moi si tu bloques ici, je t'explique le PATH en 30 secondes.
✅
Vérifie (ferme et réouvre PowerShell)
ffmpeg -versionCOPIER
💚 Tu dois voir : ffmpeg version 7.x...

ÉTAPE 05 Créer pipeline.js — le cerveau ~2 min
🧠
Ce fichier orchestre tout le système
Crée un fichier pipeline.js dans ton dossier et colle-y ce code. C'est lui que tu donneras à Claude Code pour qu'il complète et améliore :
pipeline.js — Phase 01JAVASCRIPT
// ============================================
// SYSTÈME AUTONOME — VIDÉOS AFFILIATION
// Pipeline Phase 01 : lien → brief → code
// ============================================

require('dotenv').config();
const Anthropic = require('@anthropic-ai/sdk');
const fs = require('fs-extra');
const path = require('path');

const client = new Anthropic();

// ── CONFIGURATION ──────────────────────────
const CONFIG = {
  outputDir: './output',
  model: 'claude-sonnet-4-20250514',
  videoFormat: { width: 1080, height: 1920, fps: 30 }
};

// ── ÉTAPE 1 : Analyser l'outil ─────────────
async function analyzeTool(affiliateUrl) {
  console.log('\n🔍 Analyse de l\'outil...');

  const response = await client.messages.create({
    model: CONFIG.model,
    max_tokens: 2000,
    tools: [{ type: "web_search_20250305", name: "web_search" }],
    messages: [{
      role: 'user',
      content: `Analyse cet outil à partir de son lien affilié : ${affiliateUrl}
      
Recherche et retourne en JSON :
- name: nom de l'outil
- tagline: phrase courte de description  
- mainFeatures: tableau des 3 fonctionnalités clés
- targetAudience: qui utilise cet outil
- mainBenefit: bénéfice principal en une phrase
- competitors: tableau de 2 concurrents directs
- affiliateCommission: taux de commission si trouvé
- color: couleur hex principale de la marque

Réponds UNIQUEMENT avec le JSON, sans markdown.`
    }]
  });

  const text = response.content
    .filter(b => b.type === 'text')
    .map(b => b.text).join('');

  return JSON.parse(text);
}

// ── ÉTAPE 2 : Générer le brief ─────────────
async function generateBrief(toolData, affiliateUrl) {
  console.log('\n✍️  Génération du brief vidéo...');

  const response = await client.messages.create({
    model: CONFIG.model,
    max_tokens: 4000,
    messages: [{
      role: 'user',
      content: `Tu es un expert en création de vidéos courtes virales (TikTok/Reels/Shorts).

Crée UN brief vidéo complet pour cet outil :
${JSON.stringify(toolData, null, 2)}

Lien affilié à intégrer dans le CTA : ${affiliateUrl}

Le brief doit contenir en JSON :
- videoTitle: titre accrocheur (max 60 chars)
- hook: phrase d'accroche d'ouverture (max 15 mots)
- duration: durée estimée en secondes (45-90)
- voiceScript: script voix off complet
- scenes: tableau de scènes, chacune avec :
  - id, name, startSec, endSec
  - visual: description précise de ce qu'on voit
  - animation: type d'animation Remotion (spring, interpolate, etc.)
  - remotionHint: conseil technique Remotion spécifique

Réponds UNIQUEMENT avec le JSON valide, sans markdown.`
    }]
  });

  return JSON.parse(response.content[0].text);
}

// ── ÉTAPE 3 : Générer le code Remotion ─────
async function generateRemotionCode(brief, toolData) {
  console.log('\n⚛️  Génération du code Remotion...');

  const response = await client.messages.create({
    model: CONFIG.model,
    max_tokens: 8000,
    messages: [{
      role: 'user',
      content: `Tu es un expert Remotion (librairie React pour vidéos programmatiques).

Génère le code Remotion COMPLET pour cette vidéo :

BRIEF: ${JSON.stringify(brief, null, 2)}
OUTIL: ${JSON.stringify(toolData, null, 2)}

REQUIREMENTS:
- Format : 1080x1920 (9:16 vertical), 30fps
- Un composant React par scène
- Un Root.jsx qui assemble tout dans une Composition
- Utilise spring(), interpolate(), useCurrentFrame(), useVideoConfig() de Remotion
- Animations fluides et professionnelles
- Couleur principale : ${toolData.color}
- Le lien affilié apparaît dans la dernière scène (CTA)
- Inclus les imports nécessaires dans chaque fichier

Retourne un JSON avec :
{
  "files": [
    { "path": "src/Root.jsx", "content": "..." },
    { "path": "src/scenes/Scene01.jsx", "content": "..." },
    ...
  ]
}

UNIQUEMENT le JSON valide, sans markdown.`
    }]
  });

  return JSON.parse(response.content[0].text);
}

// ── MAIN : Orchestration ───────────────────
async function run(affiliateUrl) {
  console.log('🚀 DÉMARRAGE DU PIPELINE');
  console.log(`📎 Lien affilié : ${affiliateUrl}\n`);

  await fs.ensureDir(CONFIG.outputDir);

  try {
    // 1. Analyse
    const toolData = await analyzeTool(affiliateUrl);
    console.log(`✅ Outil identifié : ${toolData.name}`);

    // 2. Brief
    const brief = await generateBrief(toolData, affiliateUrl);
    console.log(`✅ Brief généré : "${brief.videoTitle}"`);

    // 3. Code Remotion
    const remotionCode = await generateRemotionCode(brief, toolData);
    console.log(`✅ ${remotionCode.files.length} fichiers Remotion générés`);

    // 4. Sauvegarde
    const projectDir = path.join(CONFIG.outputDir, toolData.name.replace(/\s+/g, '-'));
    await fs.ensureDir(projectDir);
    await fs.writeJson(path.join(projectDir, 'tool.json'), toolData, { spaces: 2 });
    await fs.writeJson(path.join(projectDir, 'brief.json'), brief, { spaces: 2 });

    for (const file of remotionCode.files) {
      const filePath = path.join(projectDir, file.path);
      await fs.ensureDir(path.dirname(filePath));
      await fs.writeFile(filePath, file.content);
    }

    console.log(`\n🎉 PHASE 01 TERMINÉE !`);
    console.log(`📁 Projet sauvegardé : ${projectDir}`);
    console.log(`\n➡️  Prochaine étape : cd ${projectDir} && npx remotion studio`);

    return { toolData, brief, remotionCode, projectDir };

  } catch (err) {
    console.error('\n❌ Erreur :', err.message);
    throw err;
  }
}

// Lancement : node pipeline.js "https://make.com/?via=toncode"
const affiliateUrl = process.argv[2];
if (!affiliateUrl) {
  console.error('❌ Usage : node pipeline.js "https://lien-affilie.com"');
  process.exit(1);
}
run(affiliateUrl);
